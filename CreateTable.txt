create table course_table(
	CRN INT(11) NOT NULL,
    SUBJECT VARCHAR(45) NOT NULL,
    COURSENUM INT(11) NOT NULL,
    SECTION INT(11) NOT NULL,
    CAMPUS VARCHAR(45) NOT NULL,
    CREDIT VARCHAR(255),
    TITLE VARCHAR(200),
    DAYS VARCHAR(45),
    TIME VARCHAR(455),
    SEATSREM INT(11),
    WLCAP INT(11),
    WLACT INT(11),
    WLREM INT(11),
    RSVDREM INT(11),
    INSTRUCTOR VARCHAR(255),
    DATES VARCHAR(100),
    SESSION VARCHAR(45),
    LOCATION VARCHAR(45),
	PRIMARY KEY (CRN));
